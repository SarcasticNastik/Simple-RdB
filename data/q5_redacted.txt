LOAD PROJECT
ALMOST_INV <- SELECT Dnum != 1 FROM PROJECT
INV <- SELECT Dnum != 5 FROM ALMOST_INV
CRISS_CROSS <- CROSS PROJECT INV
ANS <- SELECT PROJECT_Dnum != INV_Dnum FROM CRISS_CROSS
Q5 <- PROJECT PROJECT_Pnumber,PROJECT_Dnum FROM ANS
RENAME PROJECT_Pnumber TO Pnumber FROM Q5
RENAME PROJECT_Dnum TO Dnum FROM Q5
EXPORT Q5
